96QByteArray VaultBackend::decryptData(const QByteArray &fileData, const QString &masterKey) {
97 if (fileData.size() < 72) return QByteArray();
98
99 QByteArray integrityKey = deriveKey(masterKey, fileData.mid(32, 8));
100 if (!checkIntegrity(fileData, integrityKey)) return QByteArray();
101
102 QByteArray systemID = fileData.mid(0, 32);
103 if (systemID != generateSystemHash()) return QByteArray();

209int VaultBackend::calculatePasswordStrength(const QString &password) {
210 if (password.isEmpty()) return 0;
211
212 bool hasUpper = false, hasLower = false, hasDigit = false, hasSymbol = false;
213 for (QChar c : password) {
214 if (c.isUpper()) hasUpper = true;
215 else if (c.isLower()) hasLower = true;
216 else if (c.isDigit()) hasDigit = true;
217 else hasSymbol = true;
218 }
219
220 int poolSize = 0;
221 if (hasLower) poolSize += 26;
222 if (hasUpper) poolSize += 26;
223 if (hasDigit) poolSize += 10;
224 if (hasSymbol) poolSize += 32;
225
226 // Ентропія E = L * log2(R)
227 double entropy = password.length() * std::log2(poolSize > 0 ? poolSize : 1);
228
229 int score = (int)entropy;
230 if (score > 100) score = 100;
231 return score;
232}
